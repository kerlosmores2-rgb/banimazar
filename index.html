// ==================== تسجيل الدخول ==================== document.getElementById('loginForm').onsubmit = async function(e) { e.preventDefault(); const username = document.getElementById('username').value; const password = document.getElementById('password').value; console.log('محاولة دخول:', username, password); try { // البحث عن المستخدم في جدول users const { data: users, error } = await supabase .from('users') .select('*') .eq('username', username) .eq('password', password); console.log('نتيجة البحث:', users, error); if (error) { console.error('خطأ في الاستعلام:', error); const errorDiv = document.getElementById('errorMsg'); errorDiv.textContent = 'خطأ في الاتصال بقاعدة البيانات: ' + error.message; errorDiv.classList.remove('d-none'); return; } if (users && users.length > 0) { const user = users[0]; console.log('تم تسجيل الدخول بنجاح:', user.name); localStorage.setItem('loggedUser', JSON.stringify({ id: user.id, username: user.username, name: user.name, role: user.role })); localStorage.setItem('loginTime', Date.now()); if (document.getElementById('remember').checked) { localStorage.setItem('rememberedUser', username); } window.location.href = 'dashboard.html'; } else { console.log('فشل تسجيل الدخول - المستخدم غير موجود'); const errorDiv = document.getElementById('errorMsg'); errorDiv.textContent = 'اسم المستخدم أو كلمة المرور غير صحيحة. جرب: admin / admin123'; errorDiv.classList.remove('d-none'); } } catch (err) { console.error('خطأ غير متوقع:', err); const errorDiv = document.getElementById('errorMsg'); errorDiv.textContent = 'خطأ غير متوقع: ' + err.message; errorDiv.classList.remove('d-none'); } };
